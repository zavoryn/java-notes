#### Java д»Јз ҒзӨәдҫӢпјҲ5з§Қе®һзҺ°пјү

**1. йҘҝжұүејҸпјҲEagerпјү**вҖ”вҖ”зұ»еҠ иҪҪж—¶е°ұеҲӣе»әпјҢзәҝзЁӢе®үе…ЁдҪҶеҸҜиғҪжөӘиҙ№еҶ…еӯҳ

```java
public class EagerSingleton {
    // зұ»еҠ иҪҪж—¶е°ұеҲқе§ӢеҢ–
    private static final EagerSingleton INSTANCE = new EagerSingleton();

    private EagerSingleton() {}

    public static EagerSingleton getInstance() {
        return INSTANCE;
    }
}
```

**2. жҮ’жұүејҸпјҲLazyпјү**вҖ”вҖ”з¬¬дёҖж¬Ўи°ғз”Ёж—¶жүҚеҲӣе»әпјҢйңҖеҠ еҗҢжӯҘ

```java
public class LazySingleton {
    private static LazySingleton instance;

    private LazySingleton() {}

    public static synchronized LazySingleton getInstance() {
        if (instance == null) {
            instance = new LazySingleton();
        }
        return instance;
    }
}
```

> й—®йўҳпјҡ`synchronized` й”ҒеңЁж–№жі•зә§еҲ«пјҢжҜҸж¬Ўи°ғз”ЁйғҪиҰҒеҠ й”ҒпјҢжҖ§иғҪе·®гҖӮ

**3. еҸҢйҮҚжЈҖжҹҘй”ҒпјҲDCLпјү**вҖ”вҖ”жңҖеёёз”Ёзҡ„й«ҳж•Ҳе®һзҺ°

```java
public class DCLSingleton {
    // volatile йҳІжӯўжҢҮд»ӨйҮҚжҺ’еәҸ
    private static volatile DCLSingleton instance;

    private DCLSingleton() {}

    public static DCLSingleton getInstance() {
        if (instance == null) {                    // з¬¬дёҖж¬ЎжЈҖжҹҘпјҲж— й”Ғпјү
            synchronized (DCLSingleton.class) {
                if (instance == null) {            // з¬¬дәҢж¬ЎжЈҖжҹҘпјҲжңүй”Ғпјү
                    instance = new DCLSingleton(); // в‘ еҲҶй…ҚеҶ…еӯҳ в‘ЎеҲқе§ӢеҢ– в‘ўжҢҮеҗ‘еј•з”Ё
                }
            }
        }
        return instance;
    }
}
```

> **е…ій”®йқўиҜ•зӮ№**пјҡ`volatile` дёәд»Җд№Ҳеҝ…йңҖпјҹеӣ дёә `new` ж“ҚдҪңдёҚжҳҜеҺҹеӯҗзҡ„пјҡ
> 1. еҲҶй…ҚеҶ…еӯҳз©әй—ҙ
> 2. и°ғз”Ёжһ„йҖ еҷЁеҲқе§ӢеҢ–еҜ№иұЎ
> 3. е°Ҷеј•з”ЁжҢҮеҗ‘еҶ…еӯҳең°еқҖ
>
> JVM еҸҜиғҪе°ҶжӯҘйӘӨ 2 е’Ң 3 йҮҚжҺ’еәҸпјҲжҢҮд»ӨйҮҚжҺ’пјүпјҢеҜјиҮҙеҸҰдёҖдёӘзәҝзЁӢеңЁжӯҘйӘӨ 3 еҗҺгҖҒжӯҘйӘӨ 2 еүҚжӢҝеҲ°дёҖдёӘ**жңӘеҲқе§ӢеҢ–е®Ңзҡ„еҜ№иұЎ**гҖӮ`volatile` зҰҒжӯўжҢҮд»ӨйҮҚжҺ’еәҸгҖӮ

**4. йқҷжҖҒеҶ…йғЁзұ»пјҲHolderпјү**вҖ”вҖ”еҲ©з”Ёзұ»еҠ иҪҪжңәеҲ¶пјҢе»¶иҝҹеҠ иҪҪдё”зәҝзЁӢе®үе…Ё

```java
public class HolderSingleton {
    private HolderSingleton() {}

    // йқҷжҖҒеҶ…йғЁзұ»еңЁз¬¬дёҖж¬Ўиў«еј•з”Ёж—¶жүҚеҠ иҪҪ
    private static class Holder {
        private static final HolderSingleton INSTANCE = new HolderSingleton();
    }

    public static HolderSingleton getInstance() {
        return Holder.INSTANCE;
    }
}
```

> еҺҹзҗҶпјҡJVM дҝқиҜҒдёҖдёӘзұ»зҡ„ `<clinit>()` ж–№жі•еңЁеӨҡзәҝзЁӢдёӢиў«жӯЈзЎ®еҠ й”ҒгҖӮHolder зұ»еҸӘеңЁ `getInstance()` йҰ–ж¬Ўи°ғз”Ёж—¶жүҚиў«еҠ иҪҪгҖӮ

**5. жһҡдёҫпјҲEnumпјү**вҖ”вҖ”гҖҠEffective JavaгҖӢжҺЁиҚҗзҡ„жңҖдҪіе®һи·ө

```java
public enum EnumSingleton {
    INSTANCE;

// дҪҝз”Ё

```

```java
class JdkProxyHandler implements InvocationHandler {
    private Object target;

    public JdkProxyHandler(Object target) {
        this.target = target;
    }

    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
        System.out.println("гҖҗJDKеҠЁжҖҒд»ЈзҗҶгҖ‘ејҖе§Ӣ " + method.getName());
        long start = System.currentTimeMillis();

        Object result = method.invoke(target, args);  // еҸҚе°„и°ғз”Ёзӣ®ж Үж–№жі•

        long end = System.currentTimeMillis();
        System.out.println("гҖҗJDKеҠЁжҖҒд»ЈзҗҶгҖ‘з»“жқҹ " + method.getName()
                         + "пјҢиҖ—ж—¶: " + (end - start) + "ms");
        return result;
    }

    // иҺ·еҸ–д»ЈзҗҶеҜ№иұЎ
    public static <T> T getProxy(T target) {
        return (T) Proxy.newProxyInstance(
            target.getClass().getClassLoader(),   // зұ»еҠ иҪҪеҷЁ
            target.getClass().getInterfaces(),    // жҺҘеҸЈеҲ—иЎЁ
            new JdkProxyHandler(target)           // InvocationHandler
        );
    }
}

// дҪҝз”Ё

proxy.save("еј дёү");  // proxy жҳҜ com.sun.proxy.$Proxy0 еҜ№иұЎ
```

```java
// SpringMVC дёӯпјҢDispatcherServlet дёҚзӣҙжҺҘи°ғз”Ё ControllerпјҢиҖҢжҳҜйҖҡиҝҮ HandlerAdapter

// дёҚеҗҢ Controller зҡ„жҺҘеҸЈеҗ„дёҚзӣёеҗҢпјҡ
//   - е®һзҺ° Controller жҺҘеҸЈзҡ„ж—§ејҸжҺ§еҲ¶еҷЁ
//   - дҪҝз”Ё @RequestMapping жіЁи§Јзҡ„жҺ§еҲ¶еҷЁ
//   - HttpRequestHandler
//   - Servlet

// HandlerAdapter е°ұжҳҜйҖӮй…ҚеҷЁпјҢи®© DispatcherServlet з”Ёз»ҹдёҖж–№ејҸи°ғз”ЁдёҚеҗҢзұ»еһӢзҡ„еӨ„зҗҶеҷЁ

public interface HandlerAdapter {
    // еҲӨж–ӯжҳҜеҗҰж”ҜжҢҒиҜҘеӨ„зҗҶеҷЁ
    boolean supports(Object handler);

    // з»ҹдёҖи°ғз”Ёж–№жі•
    ModelAndView handle(HttpServletRequest request,
                        HttpServletResponse response,
                        Object handler) throws Exception;
}

// е…·дҪ“йҖӮй…ҚеҷЁпјҲSpringMVC жәҗз Ғз®ҖеҢ–пјү
public class RequestMappingHandlerAdapter implements HandlerAdapter {
    @Override
    public boolean supports(Object handler) {
        return handler instanceof HandlerMethod;  // @RequestMapping жіЁи§Ј
    }

    @Override
    public ModelAndView handle(HttpServletRequest req, HttpServletResponse resp,
                               Object handler) {
        return ((HandlerMethod) handler).invokeAndHandle(req, resp);  // йҖӮй…Қи°ғз”Ё
    }
}

public class SimpleControllerHandlerAdapter implements HandlerAdapter {
    @Override
    public boolean supports(Object handler) {
        return handler instanceof Controller;  // е®һзҺ° Controller жҺҘеҸЈ
    }

    @Override
    public ModelAndView handle(HttpServletRequest req, HttpServletResponse resp,
                               Object handler) {
        return ((Controller) handler).handleRequest(req, resp);  // йҖӮй…Қи°ғз”Ё
    }
}

// DispatcherServlet ж ёеҝғеӨ„зҗҶйҖ»иҫ‘пјҲз®ҖеҢ–зүҲпјүпјҡ
public class DispatcherServlet {
    private List<HandlerAdapter> handlerAdapters;

    protected void doDispatch(HttpServletRequest req, HttpServletResponse resp) {
        Object handler = getHandler(req);                         // иҺ·еҸ–еӨ„зҗҶеҷЁ
        HandlerAdapter adapter = getHandlerAdapter(handler);     // иҺ·еҸ–йҖӮй…ҚеҷЁ
        ModelAndView mv = adapter.handle(req, resp, handler);    // йҖҡиҝҮйҖӮй…ҚеҷЁи°ғз”Ё
    }

    private HandlerAdapter getHandlerAdapter(Object handler) {
        for (HandlerAdapter adapter : handlerAdapters) {
            if (adapter.supports(handler)) return adapter;
        }
        throw new ServletException("No adapter for handler");
    }
}
```

```java
// з»„д»¶жҺҘеҸЈ
interface Coffee {
    double cost();
    String description();
}

// е…·дҪ“з»„д»¶
class SimpleCoffee implements Coffee {
    public double cost() { return 10.0; }
    public String description() { return "жҷ®йҖҡе’–е•Ў"; }
}

// иЈ…йҘ°еҷЁжҠҪиұЎзұ»
abstract class CoffeeDecorator implements Coffee {
    protected Coffee coffee;   // жҢҒжңүиў«иЈ…йҘ°зҡ„з»„д»¶

    public CoffeeDecorator(Coffee coffee) {
        this.coffee = coffee;
    }
}

// е…·дҪ“иЈ…йҘ°еҷЁпјҡеҠ зүӣеҘ¶
class MilkDecorator extends CoffeeDecorator {
    public MilkDecorator(Coffee coffee) { super(coffee); }

    public double cost() { return coffee.cost() + 2.0; }

    public String description() {
        return coffee.description() + ", еҠ зүӣеҘ¶";
    }
}

// е…·дҪ“иЈ…йҘ°еҷЁпјҡеҠ зі–
class SugarDecorator extends CoffeeDecorator {
    public SugarDecorator(Coffee coffee) { super(coffee); }

    public double cost() { return coffee.cost() + 1.0; }

    public String description() {
        return coffee.description() + ", еҠ зі–";
    }
}

// дҪҝз”ЁвҖ”вҖ”еҸҜд»Ҙж— йҷҗеҸ еҠ иЈ…йҘ°
Coffee coffee = new SimpleCoffee();
coffee = new MilkDecorator(coffee);    // еҠ зүӣеҘ¶
coffee = new SugarDecorator(coffee);   // еҶҚеҠ зі–
System.out.println(coffee.description() + " = пҝҘ" + coffee.cost());
// иҫ“еҮәпјҡжҷ®йҖҡе’–е•Ў, еҠ зүӣеҘ¶, еҠ зі– = пҝҘ13.0
```

```java
// и§ӮеҜҹиҖ…жҺҘеҸЈ
interface Observer {
    void update(String message);
}

// е…·дҪ“и§ӮеҜҹиҖ…пјҡеҸ‘йӮ®д»¶
class EmailObserver implements Observer {
    private String name;
    public EmailObserver(String name) { this.name = name; }

    @Override
    public void update(String message) {
        System.out.println("гҖҗ" + name + "-йӮ®д»¶йҖҡзҹҘгҖ‘" + message);
    }
}

// е…·дҪ“и§ӮеҜҹиҖ…пјҡеҸ‘зҹӯдҝЎ
class SmsObserver implements Observer {
    private String name;
    public SmsObserver(String name) { this.name = name; }

    @Override
    public void update(String message) {
        System.out.println("гҖҗ" + name + "-зҹӯдҝЎйҖҡзҹҘгҖ‘" + message);
    }
}

// дё»йўҳпјҲиў«и§ӮеҜҹиҖ…пјү
class OrderSubject {
    private List<Observer> observers = new ArrayList<>();
    private String status;

    // жіЁеҶҢи§ӮеҜҹиҖ…
    public void attach(Observer observer) {
        observers.add(observer);
    }

    // з§»йҷӨи§ӮеҜҹиҖ…
    public void detach(Observer observer) {
        observers.remove(observer);
    }

    // йҖҡзҹҘжүҖжңүи§ӮеҜҹиҖ…
    public void notifyObservers(String message) {
        for (Observer observer : observers) {
            observer.update(message);        // йҖҗдёӘйҖҡзҹҘ
        }
    }

    // зҠ¶жҖҒеҸҳжӣҙж—¶иҮӘеҠЁйҖҡзҹҘ
    public void setStatus(String status) {
        this.status = status;
        System.out.println("и®ўеҚ•зҠ¶жҖҒеҸҳжӣҙдёә: " + status);
        notifyObservers("и®ўеҚ•зҠ¶жҖҒе·Іжӣҙж–°дёәпјҡ" + status);  // жҺЁжЁЎејҸ
    }
}

// дҪҝз”Ё
OrderSubject order = new OrderSubject();
order.attach(new EmailObserver("еј дёү@qq.com"));
order.attach(new SmsObserver("138xxxx1234"));

// и®ўеҚ•зҠ¶жҖҒеҸҳжӣҙдёә: е·ІеҸ‘иҙ§

// гҖҗ138xxxx1234-зҹӯдҝЎйҖҡзҹҘгҖ‘и®ўеҚ•зҠ¶жҖҒе·Іжӣҙж–°дёәпјҡе·ІеҸ‘иҙ§
```

```java
// JdbcTemplate жҳҜжЁЎжқҝж–№жі•жЁЎејҸзҡ„з»Ҹе…ёеә”з”Ё
public class JdbcTemplate {
    // жЁЎжқҝж–№жі•вҖ”вҖ”е®ҡд№үдәҶжү§иЎҢ SQL зҡ„йӘЁжһ¶
    public <T> T query(String sql, RowMapper<T> rowMapper) {
        Connection conn = null;
        PreparedStatement ps = null;
        ResultSet rs = null;
        try {
            conn = dataSource.getConnection();     // 1. иҺ·еҸ–иҝһжҺҘпјҲеӣәе®ҡпјү
            ps = conn.prepareStatement(sql);       // 2. еҲӣе»ә StatementпјҲеӣәе®ҡпјү
            rs = ps.executeQuery();                // 3. жү§иЎҢ SQLпјҲеӣәе®ҡпјү
            List<T> results = new ArrayList<>();
            while (rs.next()) {
                T row = rowMapper.mapRow(rs, rs.getRow());  // 4. еӨ„зҗҶз»“жһңпјҲеҸҜеҸҳпјҒпјү
                results.add(row);
            }
            return (T) results;
        } catch (SQLException e) {
            throw new DataAccessException(e);      // 5. ејӮеёёеӨ„зҗҶпјҲеӣәе®ҡпјү
        } finally {
            close(rs);                             // 6. йҮҠж”ҫиө„жәҗпјҲеӣәе®ҡпјү
            close(ps);
            close(conn);
        }
    }
}
```

```java
    public int indexOf(Object o) {

        if (o == null) {
            while (it.hasNext())
                if (it.next() == null) return it.previousIndex();
        } else {
            while (it.hasNext())
                if (o.equals(it.next())) return it.previousIndex();
        }
        return -1;
    }
}
```

```java
                           HttpServletResponse response) throws Exception {
```

## е…ӯгҖҒйқўиҜ•ж»ЎеҲҶиҜқжңҜ

### иҜқжңҜжЁЎжқҝдёҖпјҡиҮӘжҲ‘д»Ӣз»ҚдёӯеөҢе…Ҙи®ҫи®ЎжЁЎејҸ

> "еңЁйЎ№зӣ®е®һи·өдёӯпјҢжҲ‘жіЁйҮҚи®ҫи®ЎжЁЎејҸзҡ„еә”з”ЁгҖӮжҜ”еҰӮи®ўеҚ•жЁЎеқ—з”Ё**зӯ–з•ҘжЁЎејҸ**ж¶ҲйҷӨдәҶж”Ҝд»ҳж–№ејҸзҡ„ if-else еҲҶж”Ҝпјӣе…¬е‘ҠжҺЁйҖҒз”Ё**и§ӮеҜҹиҖ…жЁЎејҸ**и§ЈиҖҰдәҶдёҚеҗҢжё йҒ“зҡ„йҖҡзҹҘйҖ»иҫ‘пјӣй…ҚзҪ®з®ЎзҗҶз”Ё**еҚ•дҫӢжЁЎејҸ**пјҲDCL + volatileпјүдҝқиҜҒе…ЁеұҖе”ҜдёҖпјӣSDK е°ҒиЈ…ж—¶з”Ё**жЁЎжқҝж–№жі•**е®ҡд№үз»ҹдёҖи°ғз”ЁжөҒзЁӢпјҢи®©дёҡеҠЎж–№еҸӘйңҖе…іжіЁдёҡеҠЎйҖ»иҫ‘гҖӮеҸҰеӨ–пјҢжҲ‘еҜ№ Spring жәҗз Ғдёӯзҡ„и®ҫи®ЎжЁЎејҸд№ҹжҜ”иҫғзҶҹжӮүпјҢжҜ”еҰӮ AOP еә•еұӮз”Ёзҡ„жҳҜ**д»ЈзҗҶжЁЎејҸ**пјҢSpringMVC зҡ„ HandlerAdapter жҳҜ**йҖӮй…ҚеҷЁжЁЎејҸ**пјҢJdbcTemplate жҳҜ**жЁЎжқҝж–№жі•жЁЎејҸ**гҖӮ"

### иҜқжңҜжЁЎжқҝдәҢпјҡиў«й—®"йЎ№зӣ®дёӯз”ЁдәҶе“Әдәӣи®ҫи®ЎжЁЎејҸ"

> "жҲ‘дё»иҰҒз”ЁеҲ°еӣӣз§Қпјҡ
> 1. **зӯ–з•ҘжЁЎејҸ**вҖ”вҖ”ж”Ҝд»ҳжЁЎеқ—ж”ҜжҢҒеҫ®дҝЎгҖҒж”Ҝд»ҳе®қгҖҒй“¶иЎҢеҚЎдёүз§Қж”Ҝд»ҳпјҢйҖҡиҝҮ Spring иҮӘеҠЁжіЁе…Ҙ `Map<String, PaymentStrategy>` е®һзҺ°зӯ–з•ҘйҖүжӢ©пјҢж–°еўһж”Ҝд»ҳж–№ејҸеҸӘйңҖеҠ зұ»дёҚж”№д»Јз ҒгҖӮ
> 2. **и§ӮеҜҹиҖ…жЁЎејҸ**вҖ”вҖ”и®ўеҚ•зҠ¶жҖҒеҸҳжӣҙеҗҺиҰҒеҗҢжӯҘеҸ‘зҹӯдҝЎгҖҒеҸ‘йӮ®д»¶гҖҒжӣҙж–°з§ҜеҲҶпјҢдҪҝз”Ё Spring EventпјҲ`ApplicationEventPublisher` + `@EventListener`пјүи§ЈиҖҰеҗ„жЁЎеқ—гҖӮ
> 3. **жЁЎжқҝж–№жі•**вҖ”вҖ”SDK е°ҒиЈ…ж—¶жҠҪиұЎзҲ¶зұ»е®ҡд№үи°ғз”ЁжөҒзЁӢпјҲеҸӮж•°ж ЎйӘҢвҶ’еүҚзҪ®еӨ„зҗҶвҶ’жү§иЎҢвҶ’еҗҺзҪ®еӨ„зҗҶвҶ’ејӮеёёеӨ„зҗҶпјүпјҢеӯҗзұ»еҸӘиҰҶеҶҷдёҡеҠЎжү§иЎҢжӯҘйӘӨгҖӮ
> 4. **иҙЈд»»й“ҫ**вҖ”вҖ”иҜ·жұӮеҸӮж•°ж ЎйӘҢз”ЁдәҶеӨҡдёӘ Validator й“ҫејҸж ЎйӘҢпјҲйқһз©әвҶ’ж јејҸвҶ’дёҡеҠЎи§„еҲҷпјүпјҢд»»дёҖеӨұиҙҘзӣҙжҺҘиҝ”еӣһгҖӮ"

### иҜқжңҜжЁЎжқҝдёүпјҡиў«й—®"JDK еҠЁжҖҒд»ЈзҗҶе’Ң CGLIB зҡ„еҢәеҲ«"

> "жҲ‘д»ҺеӣӣдёӘз»ҙеәҰеӣһзӯ”пјҡ
> 1. **еҺҹзҗҶ**пјҡJDK жҳҜеҹәдәҺеҸҚе°„+жҺҘеҸЈзҡ„пјҢз”ҹжҲҗзҡ„жҳҜе®һзҺ°зӣ®ж ҮжҺҘеҸЈзҡ„д»ЈзҗҶзұ»пјӣCGLIB жҳҜеҹәдәҺз»§жүҝзҡ„пјҢз”ҹжҲҗзҡ„жҳҜзӣ®ж Үзұ»зҡ„еӯҗзұ»гҖӮ
> 2. **йҷҗеҲ¶**пјҡJDK иҰҒжұӮзӣ®ж Үзұ»еҝ…йЎ»е®һзҺ°жҺҘеҸЈпјӣCGLIB дёҚиҰҒжұӮжҺҘеҸЈдҪҶдёҚиғҪд»ЈзҗҶ final зұ»е’Ңж–№жі•гҖӮ
> 3. **жҖ§иғҪ**пјҡJDK еҲӣе»әд»ЈзҗҶеҝ«пјҲдёҚйңҖиҰҒи§Јжһҗеӯ—иҠӮз ҒпјүпјҢдҪҶи°ғз”ЁжҳҜйҖҡиҝҮеҸҚе°„иҫғж…ўпјӣCGLIB еҲӣе»әд»ЈзҗҶж—¶йңҖиҰҒз”ҹжҲҗеӯ—иҠӮз ҒжүҖд»Ҙж…ўдёҖдәӣпјҢдҪҶи°ғз”ЁжҳҜйҖҡиҝҮ FastClass зӣҙжҺҘи°ғз”Ёеҝ«гҖӮ
> 4. **Spring дёӯзҡ„дҪҝз”Ё**пјҡSpring 4.x й»ҳи®ӨжңүжҺҘеҸЈз”Ё JDKпјҢжІЎжҺҘеҸЈз”Ё CGLIBгҖӮSpring Boot 2.x ејҖе§Ӣй»ҳи®Өз»ҹдёҖз”Ё CGLIBпјҢеӣ дёәжӣҙзҒөжҙ»пјҢдҪҶеҸҜд»ҘйҖҡиҝҮ `spring.aop.proxy-target-class=false` ж”№еӣһ JDKгҖӮ"

### иҜқжңҜжЁЎжқҝеӣӣпјҡиў«й—®"еҚ•дҫӢжЁЎејҸдёәд»Җд№Ҳ DCL иҰҒеҠ  volatile"

> "`new Singleton()` иҝҷдёӘж“ҚдҪңеңЁ JVM еұӮйқўдёҚжҳҜеҺҹеӯҗзҡ„пјҢе®ғеҲҶдёәдёүжӯҘпјҡеҲҶй…ҚеҶ…еӯҳгҖҒеҲқе§ӢеҢ–еҜ№иұЎгҖҒе°Ҷеј•з”ЁжҢҮеҗ‘еҶ…еӯҳз©әй—ҙгҖӮJVM еҸҜиғҪдјҡе°ҶеҲқе§ӢеҢ–еҜ№иұЎе’ҢжҢҮеҗ‘еј•з”ЁиҝҷдёӨжӯҘ**йҮҚжҺ’еәҸ**гҖӮеҰӮжһңжІЎжңү `volatile`пјҢеҸҜиғҪеҮәзҺ°иҝҷз§Қжғ…еҶөпјҡзәҝзЁӢ A жү§иЎҢе®Ңз¬¬дёҖгҖҒдёүжӯҘеҗҺпјҲиҝҳжІЎеҲқе§ӢеҢ–пјүпјҢзәҝзЁӢ B иҝӣе…Ҙз¬¬дёҖж¬Ў null жЈҖжҹҘпјҢеҸ‘зҺ° instance дёҚдёә nullпјҲеӣ дёәе·Із»ҸжҢҮеҗ‘дәҶеҶ…еӯҳең°еқҖпјүпјҢзӣҙжҺҘиҝ”еӣһдәҶдёҖдёӘ**иҝҳжІЎеҲқе§ӢеҢ–е®ҢжҲҗзҡ„еҜ№иұЎ**пјҢеҜјиҮҙдҪҝз”Ёж—¶жҠҘй”ҷгҖӮ`volatile` жңүдёӨдёӘдҪңз”ЁпјҡдёҖжҳҜ**зҰҒжӯўжҢҮд»ӨйҮҚжҺ’еәҸ**пјҢдәҢжҳҜ**дҝқиҜҒеҶ…еӯҳеҸҜи§ҒжҖ§**пјҢеҲҡеҘҪи§ЈеҶіиҝҷдёӘй—®йўҳгҖӮ"

### 8.4 дёҖйҒ“"жӢүйҖҡејҸ"йқўиҜ•йўҳ

> **йқўиҜ•е®ҳ**пјҡ"жҠҠдҪ зҶҹжӮүзҡ„ Spring дёӯз”ЁеҲ°зҡ„и®ҫи®ЎжЁЎејҸйғҪиҜҙдёҖиҜҙгҖӮ"

**ж»ЎеҲҶеӣһзӯ”**пјҡ

> Spring дёӯи®ҫи®ЎжЁЎејҸж— еӨ„дёҚеңЁпјҢжҲ‘иҜҙеҮ дёӘжңҖж ёеҝғзҡ„пјҡ
>
> **1. еҚ•дҫӢжЁЎејҸ**вҖ”вҖ”Spring Bean й»ҳи®ӨдҪңз”Ёеҹҹе°ұжҳҜ singletonпјҢеә•еұӮйҖҡиҝҮ `DefaultSingletonBeanRegistry` зҡ„дёүзә§зј“еӯҳпјҲ`singletonObjects`гҖҒ`earlySingletonObjects`гҖҒ`singletonFactories`пјүз®ЎзҗҶпјҢиҝҳиғҪи§ЈеҶіеҫӘзҺҜдҫқиө–гҖӮ
>
> **2. е·ҘеҺӮжЁЎејҸ**вҖ”вҖ”ж ёеҝғзҡ„ `BeanFactory` е°ұжҳҜе·ҘеҺӮж–№жі•зҡ„е®һзҺ°пјҢ`ApplicationContext` жҳҜжҠҪиұЎе·ҘеҺӮгҖӮ`FactoryBean` жҺҘеҸЈе…Ғи®ёз”ЁжҲ·иҮӘе®ҡд№ү Bean зҡ„еҲӣе»әйҖ»иҫ‘гҖӮ
>
> **3. д»ЈзҗҶжЁЎејҸ**вҖ”вҖ”AOP зҡ„е®һзҺ°еҹәзЎҖгҖӮжңүжҺҘеҸЈз”Ё JDK еҠЁжҖҒд»ЈзҗҶпјҲ`JdkDynamicAopProxy`пјүпјҢжІЎжҺҘеҸЈз”Ё CGLIBпјҲ`ObjenesisCglibAopProxy`пјүпјҢSpring Boot 2.x й»ҳи®Өз”Ё CGLIBгҖӮ
>
> **4. жЁЎжқҝж–№жі•**вҖ”вҖ”`JdbcTemplate`гҖҒ`RestTemplate`гҖҒ`RedisTemplate` е…ЁйғҪжҳҜжЁЎжқҝж–№жі•жЁЎејҸпјҢе®ҡд№үдәҶиө„жәҗиҺ·еҸ–вҶ’жү§иЎҢвҶ’йҮҠж”ҫзҡ„йӘЁжһ¶пјҢз”ЁжҲ·еҸӘйңҖе…іжіЁж ёеҝғзҡ„дёҡеҠЎйҖ»иҫ‘гҖӮ
>
> **5. йҖӮй…ҚеҷЁжЁЎејҸ**вҖ”вҖ”SpringMVC зҡ„ `HandlerAdapter` жңҖз»Ҹе…ёгҖӮи®© `DispatcherServlet` еҸҜд»Ҙз”Ёз»ҹдёҖзҡ„ж–№ејҸи°ғз”Ё `@RequestMapping` жіЁи§Јзҡ„ ControllerгҖҒе®һзҺ° `Controller` жҺҘеҸЈзҡ„ж—§ејҸ ControllerгҖҒ`HttpRequestHandler` зӯүдёҚеҗҢзұ»еһӢзҡ„еӨ„зҗҶеҷЁгҖӮ
>
> **6. иҙЈд»»й“ҫжЁЎејҸ**вҖ”вҖ”SpringMVC зҡ„ `HandlerInterceptor` жӢҰжҲӘеҷЁй“ҫпјҢиҝҳжңү AOP зҡ„еҲҮйқўжү§иЎҢй“ҫпјҲ`MethodInterceptor` зҡ„йҖ’еҪ’и°ғз”Ёй“ҫпјүгҖӮ
>
> **7. и§ӮеҜҹиҖ…жЁЎејҸ**вҖ”вҖ”Spring зҡ„ `ApplicationEvent` + `ApplicationListener`пјҲжҲ– `@EventListener`пјүпјҢж”ҜжҢҒеҗҢжӯҘе’ҢејӮжӯҘдәӢд»¶йҖҡзҹҘгҖӮ
>
> **8. зӯ–з•ҘжЁЎејҸ**вҖ”вҖ”Spring иө„жәҗеҠ иҪҪпјҲ`ResourceLoader` ж №жҚ®и·Ҝеҫ„еүҚзјҖйҖүжӢ©дёҚеҗҢзӯ–з•ҘпјүгҖҒSpring Security зҡ„и®ӨиҜҒзӯ–з•ҘгҖӮ
>
> еҰӮжһңйқўиҜ•е®ҳж„ҹе…ҙи¶ЈпјҢжҲ‘еҸҜд»Ҙе°ұд»»дҪ•дёҖдёӘеұ•ејҖи®Іжәҗз Ғе®һзҺ°гҖӮ
